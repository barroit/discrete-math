\subsection{集合の演算}

\begin{formulatopic}{$A$ と $B$ の和集合}
   A \cup B \defineas \set[x \in A \lor x \in B]{x}
\end{formulatopic}

\begin{formulatopic}{$A$ と $B$ の積集合}
   A \cap B \defineas \set[x \in A \land x \in B]{x}
\end{formulatopic}

\begin{formulatopic}{$A$ の補集合}
   \overline{A} \defineas U \setminus A
\end{formulatopic}

\begin{theorem}
   \leavevmode

   \hspace*{.5em}
   \begin{xtabular}{lll}
      ベキ等律 & $A \cup A = A$        & $A \cap A = A$ \\
      交換律   & $A \cup B = B \cup A$ & $A \cap B = B \cap A$
   \end{xtabular}

   \hspace*{.5em}
   \begin{xtabular}{ll}
      結合律 & $(A \cup B) \cup C = A \cup (B \cup C)$ \\
             & $(A \cap B) \cap C = A \cap (B \cap C)$ \\
      分配律 & $A \cup (B \cap C) = (A \cup B) \cap (A \cup C)$ \\
             & $A \cap (B \cup C) = (A \cap B) \cup (A \cap C)$
   \end{xtabular}

   \hspace*{.5em}
   \begin{xtabular}{lll}
      収集律 & $A \cup (A \cap B) = A$ & $A \cap (A \cup B) = A$
   \end{xtabular}
\end{theorem}

\begin{theorem}
   \leavevmode
   \[
      \begin{alignedat}{2}
         A \cup \varnothing &= A  \qquad A \cap \varnothing &= \varnothing\\
         A \cup U &= U            \qquad A \cap U &= A\\
         A \cup \overline{A} &= U \qquad A \cap \overline{A} &= \varnothing\\
         \overline{\overline{A}} &= A
      \end{alignedat}
   \]
\end{theorem}

\begin{theorem}
   \leavevmode

   \hspace*{.5em}
   \begin{xtabular}{l}
      $\overline{A \cup B} = \overline{A} \cap \overline{B}$ \\
      $\overline{A \cap B} = \overline{A} \cup \overline{B}$
   \end{xtabular}
\end{theorem}
