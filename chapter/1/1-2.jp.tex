\subsection{記号 $\forall$ と $\exists$}

\begin{topic}{全称記号 $\forall$}
   \begin{description}
      \item all (すべての)
      \item any (任意の)
   \end{description}

   \vspace{\parskip}
   \begin{tabular}{ll}
      記号     & $\forall x \in \mathbb{Z},\ x \in \mathbb{R}$ \\

      日本語訳 & すべての整数 $x$ に対して, $x$ は実数 \\

      英訳     & For every integer $x$, $x$ is a real number
   \end{tabular}
\end{topic}

\begin{topic}{存在記号 $\exists$}
   \begin{description}
      \item exist (存在する)
   \end{description}

   \vspace{\parskip}
   \begin{tabular}{ll}
      記号     & $\forall x \in \mathbb{Z},\ 
                  \exists y \in \mathbb{Z},\ x + y = 1$ \\

      日本語訳 & 任意の整数 $x$ に対して,
                 $x + y = 1$ となるような整数 $y$ が存在する \\

      英訳     & For every integer $x$,
                 there exists an integer $y$ such that $x + y = 1$
   \end{tabular}
\end{topic}
