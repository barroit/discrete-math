\NewDocumentEnvironment{tocpage}{}{
   \onecolumn
   \pagenumbering{roman}
}{
   \newpage
   \twocolumn
   \pagenumbering{arabic}
}

\NewDocumentEnvironment{chapter}{m}{
   \vspace{1cm}

   \begin{center}
      \LARGE \bfseries #1
   \end{center}

   \addcontentsline{toc}{part}{#1}
}{
   \setcounter{section}{0}
}

\NewDocumentEnvironment{topic}{o}{
   \vspace{\parskip}
   \IfValueT{#1}{\textbf{#1}}
   \list{}{\leftmargin=1em}
      \item\relax
}{
   \endlist
}

\NewDocumentEnvironment{formulatopic}{m}{
   \vspace{\parskip}
   \textbf{#1}
   \[
}{
   \]
}

\NewDocumentEnvironment{itemtopic}{o}{
   \IfValueT{#1}{\textbf{#1}}
   \list{}{\leftmargin=1em}
}{
   \endlist
}

\NewDocumentEnvironment{xtabular}{m}{
   \hspace*{-\tabcolsep}\begin{tabular}{#1}
}{
   \end{tabular}
}
